10  /**
11   * {@link ExceptionUtils}クラスのテストを実行します。
12   * @author ycookjp my project
13   */
14  public class ExceptionUtilsTest {
15      /** Logger. */
16      private static Logger logger = LoggerFactory.getLogger(ExceptionUtilsTest.class);
17  
18      /**
19       * {@link ExceptionUtils#getStackTraceString(Exception)}メソッドのテストをします。
20       * <ul>
21       * <li>
22       *   引数に{@link NullPointerException}を指定すると、戻り値の文字列の長さは、
23       *   「例外のtoString()の長さ＋このクラスの完全修飾名の長さ＋メソッド名の長さ」を
24       *   超えること
25       * </li>
26       * <li>引数にnullを指定するとnullを返すこと</li>
27       * </ul>
28       */
29      @Test
30      public void testGetStackTraceString() {
31          logger.debug("***** BEGIN testGetStackTraceString");
32          // NullPointerExceptionを発生させる
33          String str = null;
34          try {
35              str.toLowerCase();
36          } catch (Exception e) {
37              String stackTrace = ExceptionUtils.getStackTraceString(e);
38              logger.debug("ExceptionUtils.getStackTraceString(e)\n    ==> {}", stackTrace);
39              Assert.assertTrue("引数にNullPointerExceptionを指定すると、戻り値の文字列の長さは、「例外のtoString()の長さ＋このクラスの完全修飾名の長さ＋メソッド名の長さ」を超えること",
40                      stackTrace.length() > e.toString().length() + ExceptionUtilsTest.class.getName().length()
41                      + "testGetStackTraceString".length());
42  
43              stackTrace = ExceptionUtils.getStackTraceString(null);
44              logger.debug("ExceptionUtils.getStackTraceString(null) ==> {}", stackTrace);
45              Assert.assertNull("引数にnullを指定するとnullを返すこと", stackTrace);
46          } finally {
47              logger.debug("***** END testGetStackTraceString");
48          }
49      }
50  }
