13  /**
14   * {@link PropertiesUtils}クラスのテストを実行します。
15   * @author ycookjp my project
16   */
17  public class PropertiesUtilsTest {
18      /** Logger. */
19      private Logger logger = LoggerFactory.getLogger(PropertiesUtilsTest.class);
20      /**
21       * {@link PropertiesUtils#loadResource(String)}メソッドのテストを実行します。
22       * <ul>
23       * <li>ASCII文字の値「value1」を取得できること</li>
24       * <li>日本語の値「値その２」を取得できること</li>
25       * <li>引数に存在しないリソースを指定すると{@link IOException}が発生すること</li>
26       * <li>引数にnullを指定するとnullが返ること</li>
27       * </ul>
28       */
29      @Test
30      public void testLoadResource() {
31          logger.debug("***** START testLoadResource *****");
32          try {
33              logger.debug("PropertiesUtils.loadResource(\"test/myproject/java/utils/testprop.properties\")");
34              Properties prop = PropertiesUtils.loadResource(
35                      "test/myproject/java/utils/testprop.properties");
36  
37              logger.debug("prop.get(\"key1\") ==> {}", prop.get("key1"));
38              Assert.assertEquals("ASCII文字の値「value1」を取得できること",
39                      "value1", prop.get("key1"));
40  
41              logger.debug("prop.get(\"key2\") ==> {}", prop.get("key2"));
42              Assert.assertEquals("日本語の値「値その２」を取得できること",
43                      "値その２", prop.get("key2"));
44  
45              try {
46                  prop = PropertiesUtils.loadResource("/test/myproject/java/utils/testprop.properties");
47                  Assert.fail("存在しないリソースを指定すると{@link IOException}が発生すること：例外が発生しない");
48              } catch (Exception e) {
49                  if (!(e instanceof IOException)) {
50                      Assert.fail("存在しないリソースを指定すると{@link IOException}が発生すること：IOException以外の例外が発生");
51                  }
52              }
53  
54              prop = PropertiesUtils.loadResource(null);
55              Assert.assertNull("引数にnullを指定するとnullが返ること", prop);
56          } catch (IOException ioe) {
57              Assert.fail(ioe.toString());
58          } catch (Exception e) {
59              Assert.fail(e.toString());
60          } finally {
61              logger.debug("***** END testLoadResource *****");
62          }
63      }
64  
65      /**
66       * {@link myproject.java.utils.PropertiesUtils#replaceKeyword(Properties)}メソッドのテストを実行します。
67       * <ul>
68       * <li>値の「${key3}」はkey3の値に変換されること</li>
69       * <li>「${java.runtime.name}」はシステムプロパティ「java.runtime.name」の値に返還されること</li>
70       * <li>引数にnullを指定しても例外が発生しないこと</li>
71       * </ul>
72       */
73      @Test
74      public void testReplaceKeyword() {
75          logger.debug("***** BEGIN testReplaceKeyword *****");
76          try {
77              logger.debug("PropertiesUtils.loadResource(\"test/myproject/java/utils/testprop.properties\")");
78              Properties prop = PropertiesUtils.loadResource("test/myproject/java/utils/testprop.properties");
79              PropertiesUtils.replaceKeyword(prop);
80  
81              logger.debug("prop.getProperty(\"key4\") ==> {}", prop.getProperty("key4"));
82              Assert.assertEquals("値の「${key3}」はkey3の値に変換されること",
83                      "key3の値は\"" + prop.getProperty("key3") + "\"です", prop.getProperty("key4"));
84  
85              logger.debug("prop.getProperty(\"key5\") ==> {}", prop.getProperty("key5"));
86              Assert.assertEquals("「${java.runtime.name}」はシステムプロパティ「java.runtime.name」の値に返還されること",
87                      "java.runtime.name=" + System.getProperty("java.runtime.name"), prop.getProperty("key5"));
88  
89              try {
90                  PropertiesUtils.replaceKeyword(null);
91                  // OK
92              } catch (Exception e) {
93                  Assert.fail("引数にnullを指定しても例外が発生しないこと：例外が発生");
94              }
95          } catch (IOException ioe) {
96              Assert.fail(ioe.toString());
97          } catch (Exception e) {
98              Assert.fail(e.toString());
99          } finally {
100             logger.debug("***** END testReplaceKeyword *****");
101         }
102     }
