24      /**
25       * クラスを指定して、そのクラスが配置されているクラスパスのディレクトリの親ディレクトリ、
26       * またはjarファイルのパスの親ディレクトリを取得します。
27       * @param cls クラスを指定します。
28       * @return 取得したディレクトリの{@link java.io.File}オブジェクトを返します。
29       *      ディレクトリを取得できなかった場合（主にセキュリティの製薬による）はnullを返します。
30       * @throws URISyntaxException URIの構文が間違っている場合
31       */
32      public static File getClassDirParent(Class<? extends Object> cls) throws URISyntaxException {
33          if (cls == null) {
34              return null;
35          }
36  
37          // 指定されたクラスが配置されたクラスパスのディレクトリまたはJarファイルのパスを取得
38          CodeSource cs = cls.getProtectionDomain().getCodeSource();
39  
40          if (cs != null) {
41              // CodeSourceが取得できた場合
42              URL location = cs.getLocation();
43              URI clsUri = location.toURI();
44  
45              File clsDirPath = new File(clsUri.getPath());
46  
47              // 取得したファイルの親ディレクトリを返す。
48              File clsDirParent = clsDirPath.getParentFile();
49  
50              return clsDirParent;
51          } else {
52              // CoudeSourceが取得できなかった場合 - 主にセキュリティ的な理由による
53              return null;
54          }
55      }

57      /**
58       * フィールド名を指定して、クラスで定義されているフィールドを取得します。
59       * <p>
60       * クラス定義されているフィールドの中で、指定された名前と一致するものを返します。フィールドの
61       * スコープはpublic以外のものでも名前が一致すればそのフィールドを返します。指定されたクラスに
62       * フィールドが見つからなかった場合は、以下の手順でインターフェース及びスーパー・クラスを
63       * 再帰的に取得して指定されたフィールド名に一致するフィールドを探索します。
64       * <ol>
65       * <li>
66       *   指定されたクラスを実装しているインターフェースを順番に取得して、そのインターフェースで
67       *   定義されているフィールドを検索する
68       * </li>
69       * <li>
70       *   指定されたクラスのスーパー・クラスを取得して、そのクラスで定義されているフィールドを
71       *   検索する。
72       * </li>
73       * </ol>
74       *
75       * @param classz クラス
76       * @param name フィールド名
77       * @param modifiers フィールドの取得対象のアクセス修飾子を{@link java.lang.reflect.Modifier}クラスの定数の和で
78       *      指定する。取得対象のアクセス修飾子を指定しない場合は、-1を指定すること。
79       * @return 見つかったフィールドを返します。クラスまたはフィールド名にnullを指定した場合は
80       *      nullを返します。
81       * @throws NoSuchFieldException 指定された名前と一致するフィールドが見つからなかった場合
82       * @throws SecurityException セキュリティ・マネージャでエラーが発生した場合。エラーの
83       *      発生する条件は、{@link Class#getDeclaredField(String)}の仕様を参照のこと。
84       */
85      public static Field findField(Class<?> classz, String name, int modifiers)
86              throws NoSuchFieldException, SecurityException {
87          if (classz == null || name == null) {
88              return null;
89          }
90  
91          Field field = null;
92          try {
93              field = classz.getDeclaredField(name);
94          } catch (NoSuchFieldException e) {
95              for (Class<?> iface : classz.getInterfaces()) {
96                  try {
97                      field = findField(iface, name, modifiers);
98                      break;
99                  } catch (NoSuchFieldException ee) { }
100             }
101 
102             if (field == null) {
103                 Class<?> superClass = classz.getSuperclass();
104                 if (superClass != null) {
105                     field = findField(superClass, name, modifiers);
106                 } else {
107                     throw e;
108                 }
109             }
110         }
111 
112         if (modifiers > 0 && (field.getModifiers() & modifiers) == 0) {
113             throw new NoSuchFieldException(classz.getName() + "." + name + "does not match modifiers");
114         }
115 
116         return field;
117     }

120      * メソッド名とパラメータの型を指定して、クラスで定義されているフィールドを取得します。
121      * <p>
122      * クラス定義されているメソッドの中で、指定された名前及びパラメータの型一致するものを
123      * 返します。メソッドのスコープはpublic以外のものでも名前と型が一致すればそのメソッドを
124      * 返します。指定されたクラスにメソッドが見つからなかった場合は、以下の手順で
125      * インターフェース及びスーパー・クラスを再帰的に取得して指定されたメソッド名及び型に
126      * 一致するメソッドを探索します。
127      * <ol>
128      * <li>
129      *   指定されたクラスを実装しているインターフェースを順番に取得して、そのインターフェースで
130      *   定義されているメソッドを検索する
131      * </li>
132      * <li>
133      *   指定されたクラスのスーパー・クラスを取得して、そのクラスで定義されているメソッドを
134      *   検索する。
135      * </li>
136      * </ol>
137      *
138      * @param classz クラス
139      * @param name メソッド名
140      * @param parameterTypes パラメータの型の配列。メソッドのパラメータの順番にパラメータの
141      *      クラスを配列に設定する。
142      * @param modifiers フィールドの取得対象のアクセス修飾子を{@link java.lang.reflect.Modifier}クラスの定数の和で
143      *      指定する。取得対象のアクセス修飾子を指定しない場合は、-1を指定すること。
144      * @return 見つかったメソッドを返します。クラス名またはメソッド名にnullを指定した場合はnullを
145      *      返します。
146      * @throws NoSuchMethodException 指定された名前及び方に一致するメソッドが見つからなかった場合
147      * @throws SecurityException セキュリティ・マネージャでエラーが発生した場合。エラーの
148      *      発生する条件は、{@link Class#getDeclaredMethod(String, Class...)}の仕様を参照のこと。
149      */
150     public static Method findMethod(Class<?> classz, String name, Class<?>[] parameterTypes, int modifiers)
151             throws NoSuchMethodException, SecurityException {
152         if (classz == null || name == null) {
153             return null;
154         }
155 
156         Method method = null;
157         try {
158             method = classz.getDeclaredMethod(name, parameterTypes);
159         } catch (NoSuchMethodException e) {
160             for (Class<?> iface : classz.getInterfaces()) {
161                 try {
162                     method = findMethod(iface, name, parameterTypes, modifiers);
163                     break;
164                 } catch (NoSuchMethodException ee) { }
165             }
166 
167             if (method == null) {
168                 Class<?> superClass = classz.getSuperclass();
169                 if (superClass != null) {
170                     method = findMethod(superClass, name, parameterTypes, modifiers);
171                 } else {
172                     throw e;
173                 }
174             }
175         }
176 
177         if (modifiers > 0 && (method.getModifiers() & modifiers) == 0) {
178             if (parameterTypes == null) {
179                 parameterTypes = new Class[] {};
180             }
181             throw new NoSuchMethodException(classz.getName() + "." + name + "(" + Arrays.asList(parameterTypes)
182                     + ") does not match modifiers");
183         }
184 
185         return method;
186     }
