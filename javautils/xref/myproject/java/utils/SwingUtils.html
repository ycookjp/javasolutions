21      /**
22       * {@link Component}の名前を指定して、{@link Container}に保持されている{@link Component}を
23       * 取得します。引数で指定された{@link Container}が保持する{@link Component}から指定された
24       * 名前のものが見つからなかった場合は、再帰的に{@link Component}を検索します。
25       * <dl>
26       * <dt>JFrameからComponentを取得する例</dt>
27       * <dd>
28       *   JFrameの中に保持されている{@link Component}を取得する場合は、{@link Container}に
29       *   JFrameのcontentPainを指定して、{@link Component}を取得します。
30       *   <table border="1"><caption>コードの例</caption><tr><td><pre>
31       * JFrame frame = new JFrame();
32       * ...
33       * Component component = SwingUtils.getComponentByName(frame.getComponentPane(), "MyComponentName");
34       * ...
35       *   </pre></td></tr></table>
36       * </dd>
37       * </dl>
38       * @param container 取得対象の{@link Component}を保持している{@link Container}
39       * @param name 検索対象の{@link Container}の名前
40       * @return 取得した{@link Container}のインスタンスを返します。
41       *     引数container、またはnameにnullが指定された場合はnullを返します。
42       *     指定された名前の{@link Container}が見つからなかった場合はnullを返します。
43       */
44      public static Component getComponentByName(Container container, String name) {
45          Component result = null;
46          if (container == null || name == null) {
47              return null;
48          }
49  
50          Component[] components = container.getComponents();
51          for (Component component : components) {
52              String componentName = component.getName();
53              if (name.equals(componentName)) {
54                  // 指定された名前のComponentが見つかったらそのコンポーネントを返す。
55                  result = component;
56                  break;
57              }
58          }
59  
60          if (result == null) {
61              // Container直下のComponentから見つからなかった場合は、再帰的にComponentを取得する
62              for (Component component  : components) {
63                  if (component instanceof Container) {
64                      Container childContainer = (Container) component;
65                      Component child = getComponentByName(childContainer, name);
66                      if (child != null && name.equals(child.getName())) {
67                          result = child;
68                          break;
69                      }
70                  }
71              }
72          }
73  
74          return result;
75      }
