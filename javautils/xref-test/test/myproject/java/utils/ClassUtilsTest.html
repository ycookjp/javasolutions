36      /**
37       * {@link ClassUtils#getClassDirParent(Class)}メソッドのテストを実行します。
38       * <ul>
39       * <li>引数に{@link ClassUtilsTest}(テストクラス)を指定すると、結果が/targetで終わる</li>
40       * <li>引数に{@link org.junit.Test}(Jarのクラス)を指定すると、結果に"/junit/junit/"が含まれる</li>
41       * <li>引数に{@link java.lang.StackOverflowError}のクラスを指定するとnullが返る</li>
42       * <li>引数にnullを設定するとnullが返る</li>
43       * </ul>
44       */
45      @Test
46      public void testGetClassFilePath() {
47          logger.debug("***** BEGIN testGetClassFilePath");
48          try {
49              File file = ClassUtils.getClassDirParent(ClassUtilsTest.class);
50              String absolutePath = file.getAbsolutePath();
51              logger.debug("ClassUtils.getClassDirParent(TestClassUtils.class) ==> {}", absolutePath);
52              Assert.assertEquals("引数にTestClassUtils(テストクラス)を指定すると、結果が/targetで終わる",
53                      "/target", absolutePath.substring(absolutePath.length() - "/target".length()));
54  
55              file = ClassUtils.getClassDirParent(org.junit.Test.class);
56              absolutePath = file.getAbsolutePath();
57              logger.debug("ClassUtils.getClassDirParent(org.junit.Test.class) ==> {}", absolutePath);
58              Assert.assertTrue("引数にorg.junit.Test(Jarのクラス)を指定すると、結果に\"/junit/junit/\"が含まれる",
59                      absolutePath.indexOf("/junit/junit/") > 0);
60  
61              file = ClassUtils.getClassDirParent(java.lang.StackOverflowError.class);
62              logger.debug("ClassUtils.getClassDirParent(java.lang.StackOverflowError.class) ==> {}", file);
63              Assert.assertNull("java.lang.StackOverflowErrorのクラスを指定するとnullが返る", file);
64  
65              file = ClassUtils.getClassDirParent(null);
66              logger.debug("ClassUtils.getClassDirParent(null) ==> {}", file);
67              Assert.assertNull("引数にnullを設定するとnullが返る", file);
68          } catch (URISyntaxException use) {
69              use.printStackTrace();
70              Assert.fail(use.toString());
71          } catch (Exception e) {
72              System.out.println(ExceptionUtils.getStackTraceString(e));
73              Assert.fail(e.toString());
74          } finally {
75              logger.debug("***** END testGetClassFilePath");
76          }
77      }
