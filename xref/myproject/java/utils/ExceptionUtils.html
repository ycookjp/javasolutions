3   /**
4    * {@link java.lang.Exception}の操作を提供します。

6    * Copyright ycookjp

8    */
9   public class ExceptionUtils {
10      /**
11       * 外部空のインスタンス化を抑止するためのコンストラクタ。
12       */
13      private ExceptionUtils() { }
14  
15      /**
16       * {@link java.lang.Exception}のスタックトレースを文字列に変換します。
17       * @param e 例外
18       * @return スタックトレースを文字列に変換した結果を返します。引数 e にnullを指定すると
19       *      nullを返します。
20       */
21      public static String getStackTraceString(Exception e) {
22          if (e == null) {
23              return null;
24          }
25  
26          StringBuilder sb = new StringBuilder();
27  
28          sb.append(e.toString());
29          sb.append(System.getProperty("line.separator"));
30          StackTraceElement[] elements = e.getStackTrace();
31          for (StackTraceElement element : elements) {
32              sb.append("    at " + element.toString());
33              sb.append(System.getProperty("line.separator"));
34          }
35  
36          return sb.toString();
37      }
38  }
